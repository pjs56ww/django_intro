{% endif %} {% endfor %}

3. length filter 활용

{% for message in messages %} {% if message|length > 5 %}

{{ message }}, 글자가 너무 길어요.

{% else %}

{{ message }}, {{ message|length }}

{% endif %} {% endfor %}
